""" 写个简单统计运行时间的函数
"""

import time

def measure_time(func):

    def wrapped(*args, **kwargs):
        start = time.time()

        try:
            return func(*args, **kwargs)
        finally:
            runtime = time.time() - start
            print("Execution time: %.6f seconds" % runtime)

    return wrapped

@measure_time
def f(n):
    l = []
    for i in range(n):
        l.append(i)

if __name__ == "__main__":
    f(1000000)

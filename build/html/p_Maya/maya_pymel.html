fileName = pm.sceneName().splitext()[0].basename() startFrame = pm.env.minTime endFrame = pm.env.maxTime cameraName = “%s_f%03d_f%03d” % (fileName, startFrame, endFrame) cameraList = [cam for cam in pm.listCameras(p = True)] selPanel = pm.getPanel(withFocus = True) selCam = pm.modelPanel(selPanel, q=True, camera=True) cam = pm.PyNode(selCam) cam.rename(cameraName)

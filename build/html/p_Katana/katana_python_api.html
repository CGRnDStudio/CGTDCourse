批量修改某一类型节点参数

import NodegraphAPI

print(dir(NodegraphAPI))

nodes = NodegraphAPI.GetAllNodesByType("Alembic_In")

for node in nodes:
    print(node)
    parm = node.getParameter("abcAsset")
    oldPath = parm.getValue(0)
    newPath = oldPath.replace("D:/cache/", "Z:/cache/")
    parm.setValue(newPath, 0)
